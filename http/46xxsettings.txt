############################################################
## Avaya 96x1 / 9641GS SIP settings for Zadarma PBX
############################################################

## VLAN layout:
## - Phone/voice traffic tagged on VLAN 70.
## - PC port traffic remains untagged and lands on the switch native VLAN 1.
SET L2Q 1
SET L2QVLAN 70
SET L2QAUD 6
SET L2QSIG 6
SET VLANSEP 1
SET VLANSEPMODE 0
SET PHY2VLAN 0
SET PHY2PRIO 0
SET VLANTEST 0

## SIP server / registrar
SET SIPDOMAIN pbx.zadarma.com
SET SIPPORT 5060
SET SIP_CONTROLLER_LIST 192.168.80.10:5060;transport=udp
SET SIPREGPROXYPOLICY alternate
SET SIPSIGNAL 0
SET ENABLE_UDP_TRANSPORT 1

## Open SIP mode, with a local shim for the Avaya PPM requests this firmware still makes.
SET DISCOVER_AVAYA_ENVIRONMENT 1
SET ENABLE_AVAYA_ENVIRONMENT 0
SET ENABLE_IPOFFICE 0
SET ENABLE_3PCC_ENVIRONMENT 1
SET ENABLE_PPM_SOURCED_SIPPROXYSRVR 0
SET CONFIGURATION_SERVER 192.168.80.10
SET CONFIG_SERVER_SECURE_MODE 0
SET SIMULTANEOUS_REGISTRATIONS 1
SET CONTROLLER_SEARCH_INTERVAL 16
SET ASTCONFIRMATION 16
SET SIG 2

## Zadarma PBX extension login
## The forced login is loaded from the phone-specific MAC file below.
SET ENABLE_SIP_USER_ID 0

## Registration / NAT friendliness
SET REGISTERWAIT 120
SET RECOVERYREGISTERWAIT 60
SET WAIT_FOR_REGISTRATION_TIMER 60
SET WAIT_FOR_UNREGISTRATION_TIMER 32
SET CONNECTION_REUSE 1
SET SYMMETRIC_RTP 1
SET ENABLE_EARLY_MEDIA 1

## DTMF and voicemail/MWI
SET SEND_DTMF_TYPE 2
SET DTMF_PAYLOAD_TYPE 101
SET SUBSCRIBE_SECURITY 0
SET SUBSCRIBE_LIST_NON_AVAYA ""
SET MWISRVR ""

## Audio codecs
SET ENABLE_G711U 1
SET ENABLE_G711A 1
SET ENABLE_G722 0
SET ENABLE_G726 0
SET ENABLE_G729 0

## Timing
SET NO_DIGITS_TIMEOUT 10
SET INTER_DIGIT_TIMEOUT 4
SET SNTPSRVR pool.ntp.org
SET GMTOFFSET "-3:00"
SET DSTOFFSET 0
SET TIMEZONE America/Argentina/Buenos_Aires
SET DATEFORMAT %d/%m/%y
SET TIMEFORMAT 2
SET DAYLIGHT_SAVING_SETTING_MODE 0

## Local phone UI
SET PHNNUMOFSA 3
SET SPEAKERSTAT 2
SET DISPLAY_NAME_NUMBER 1
SET LOGOS LTM=http://192.168.80.10/ltm-logo-232x140.jpg
SET CURRENT_LOGO LTM

## Temporary troubleshooting logs
SET LOGSRVR 192.168.80.10
SET SYSLOG_ENABLED 1
SET SYSLOG_LEVEL 7
SET LOCAL_LOGS_ENABLED 1
SET LOCAL_LOG_LEVEL 7
SET LOG_CATEGORY CONFIG,PPMDATA,PPMMESSAGE,ALSIP,SESSION,CORE,UI

## Deskphone SIP 7.1.1+ requests a per-device file for automatic login.
GET $MACADDR.txt
